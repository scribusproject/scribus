INCLUDE_DIRECTORIES(
${CMAKE_SOURCE_DIR}
${CMAKE_SOURCE_DIR}/scribus
)

SET(EXPORTSVG_PLUGIN_MOC_CLASSES
svgexplugin.h
)

SET(EXPORTSVG_PLUGIN_SOURCES
svgexplugin.cpp
)

SET(SCRIBUS_EXPORTSVG_PLUGIN "svgexplugin")

#IF(QT4_WRAP_CPP)
  QT4_WRAP_CPP(EXPORTSVG_PLUGIN_MOC_SOURCES ${EXPORTSVG_PLUGIN_MOC_CLASSES})
#ENDIF(QT4_WRAP_CPP)

ADD_LIBRARY(${SCRIBUS_EXPORTSVG_PLUGIN} MODULE ${EXPORTSVG_PLUGIN_SOURCES} ${EXPORTSVG_PLUGIN_MOC_SOURCES})

#Select Libart_LGPL or Cairo
IF(RENDER_LIB STREQUAL "LIBART")
  INCLUDE_DIRECTORIES(${LIBART_LGPL_INCLUDE_DIR})
  TARGET_LINK_LIBRARIES(${SCRIBUS_EXPORTSVG_PLUGIN} ${LIBART_LGPL_LIBRARIES})
ENDIF(RENDER_LIB STREQUAL "LIBART")
IF(RENDER_LIB STREQUAL "CAIRO")
  INCLUDE_DIRECTORIES(${CAIRO_INCLUDE_DIR})
  TARGET_LINK_LIBRARIES(${SCRIBUS_EXPORTSVG_PLUGIN} ${CAIRO_LIBRARIES})
  LINK_DIRECTORIES(${PREFIX_CAIRO_LIB})
ENDIF(RENDER_LIB STREQUAL "CAIRO")


INSTALL(TARGETS ${SCRIBUS_EXPORTSVG_PLUGIN}
  LIBRARY
  DESTINATION ${PLUGINDIR}
  PERMISSIONS ${PLUGIN_PERMISSIONS}
)
